# Задача 2. Двусвязный список.

## Условие
Реализуйте структуру данных «Двусвязный список» на языке программирования по вашему выбору. Необходимо реализовать создание пустого списка, добавление элемента списка, удаление элемента, поиск элемента по номеру и по значению. Желательно использовать ООП. По возможности обеспечить все возможные проверки.
## Решение
### Идея
Узел `DNode` хранит `prev` и `next`. Список держит `head`, `tail`, `_size`. Двусторонние ссылки дают $O(1)$ для `append`/`prepend`, удаления по ссылке на узел и обхода в обе стороны. Поиск по индексу оптимизирован: идём от ближнего конца — до $O(n/2)$ шагов.
### Код

```python
class DNode:
    __slots__ = ("value", "prev", "next")
    def __init__(self, v, p=None, n=None):
        self.value, self.prev, self.next = v, p, n

class DoublyLinkedList:
    def __init__(self):
        self.head = self.tail = None
        self._size = 0

    def __len__(self): return self._size
    
    def __iter__(self):
        cur = self.head
        while cur is not None:
            yield cur.value
            cur = cur.next

    def append(self, v):
        node = DNode(v, self.tail, None)
        if self.tail is None: self.head = self.tail = node
        else: self.tail.next = node; self.tail = node
        self._size += 1

    def _node_at(self, i):
        if not (0 <= i < self._size): raise IndexError
        if i <= self._size // 2:
            cur = self.head
            for _ in range(i): cur = cur.next
        else:
            cur = self.tail
            for _ in range(self._size - 1 - i): cur = cur.prev
        return cur

    def insert(self, i, v):
        if not (0 <= i <= self._size): raise IndexError
        if i == self._size: self.append(v); return
        nxt = self._node_at(i)
        node = DNode(v, nxt.prev, nxt)
        if nxt.prev is None: self.head = node
        else: nxt.prev.next = node
        nxt.prev = node
        self._size += 1

    def _detach(self, node):
        if node.prev: node.prev.next = node.next
        else:         self.head = node.next
        if node.next: node.next.prev = node.prev
        else:         self.tail = node.prev
        self._size -= 1

    def remove_at(self, i):  self._detach(self._node_at(i))

    def remove_value(self, v):
        cur = self.head
        while cur:
            if cur.value == v:
                self._detach(cur); return True
            cur = cur.next
        return False

    def find_index(self, v):
        for i, x in enumerate(self):
            if x == v: return i
        return -1

    def get(self, i): return self._node_at(i).value
```
### Сложности
- `append` (есть `tail`) — $O(1)$.
- `insert(i,v)`, `remove_at(i)`, `get(i)` — $O(\min(i, n-i))$ в среднем $O(n/4)$, в худшем $O(n/2)$.
- `remove_value`, `find_index` — $O(n)$.
- При наличии прямой ссылки на узел удаление — $O(1)$ (главное преимущество двусвязного списка).
