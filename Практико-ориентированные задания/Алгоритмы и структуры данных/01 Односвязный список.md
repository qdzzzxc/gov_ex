# Задача 1. Односвязный список.

## Условие
Реализуйте структуру данных «Односвязный список» на языке программирования по вашему выбору. Необходимо реализовать создание пустого списка, добавление элемента списка, удаление элемента, поиск элемента по номеру и по значению. Желательно использовать ООП. По возможности обеспечить все возможные проверки.

## Решение

### Программа минимум

```python
class Node:
    def __init__(self, v, nxt=None): self.v, self.next = v, nxt  # узел: значение + ссылка

class LinkedList:
    def __init__(self): self.head = None                         # пустой список
    def append(self, v):                                         # добавление в конец
        if not self.head: self.head = Node(v); return
        cur = self.head
        while cur.next: cur = cur.next
        cur.next = Node(v)
    def get(self, i):                                            # поиск по номеру
        cur = self.head
        for _ in range(i): cur = cur.next
        return cur.v
    def find(self, v):                                           # поиск по значению
        cur, i = self.head, 0
        while cur:
            if cur.v == v: return i
            cur, i = cur.next, i + 1
        return -1
    def remove(self, v):                                         # удаление первого значения
        prev, cur = None, self.head
        while cur:
            if cur.v == v:
                if prev: prev.next = cur.next
                else: self.head = cur.next
                return True
            prev, cur = cur, cur.next
        return False
```

### Полное решение

### Идея
Узел `Node` хранит значение и ссылку `next` на следующий узел. Список держит только ссылку на голову `head` и размер `_size` (чтобы `len` был $O(1)$). Все операции реализуются обходом от головы.

### Код

```python
class Node:
    __slots__ = ("value", "next")
    def __init__(self, value, nxt=None):
        self.value = value
        self.next = nxt

class SinglyLinkedList:
    def __init__(self):
        self.head = None
        self._size = 0

    def __len__(self):
        return self._size

    def __iter__(self):
        cur = self.head
        while cur is not None:
            yield cur.value
            cur = cur.next

    def _node_at(self, index):
        if not (0 <= index < self._size):
            raise IndexError("index out of range")
        cur = self.head
        for _ in range(index):
            cur = cur.next
        return cur

    def append(self, value):
        node = Node(value)
        if self.head is None:
            self.head = node
        else:
            cur = self.head
            while cur.next is not None:
                cur = cur.next
            cur.next = node
        self._size += 1

    def insert(self, index, value):
        if not (0 <= index <= self._size):
            raise IndexError("index out of range")
        if index == 0:
            self.head = Node(value, self.head)
        else:
            prev = self._node_at(index - 1)
            prev.next = Node(value, prev.next)
        self._size += 1

    def remove_at(self, index):
        if not (0 <= index < self._size):
            raise IndexError("index out of range")
        if index == 0:
            self.head = self.head.next
        else:
            prev = self._node_at(index - 1)
            prev.next = prev.next.next
        self._size -= 1

    def remove_value(self, value):
        prev, cur = None, self.head
        while cur is not None:
            if cur.value == value:
                if prev is None:
                    self.head = cur.next
                else:
                    prev.next = cur.next
                self._size -= 1
                return True
            prev, cur = cur, cur.next
        return False

    def find_index(self, value):
        for i, v in enumerate(self):
            if v == value:
                return i
        return -1

    def get(self, index):
        return self._node_at(index).value
```

### Прогон

```
sll = [10, 20, 30, 40]; insert(2, 25) → [10, 20, 25, 30, 40], len = 5
get(3) = 30
find(40) = 4
remove_value(20); remove_at(0) → [25, 30, 40]
```

### Сложности
- `append`, `get`, `insert(i, v)`, `remove_at(i)`, `find_index`, `remove_value` — $O(n)$ в худшем случае (требуют обхода).
- `len` — $O(1)$ за счёт хранения `_size`.
- `__iter__` — $O(n)$ суммарно, $O(1)$ на шаг.

### Замечания
- Все методы проверяют диапазон индексов и явно поднимают `IndexError`.
- `remove_value` снимает только первое совпадение (типичное поведение).
- Если нужен $O(1)$-`append`, можно дополнительно хранить ссылку на хвост `tail`.
