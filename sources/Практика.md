class UnorderedList:

  def \_\_init\_\_\(self\):

    self\.head = None

    self\.tail = None

  def isEmpty\(self\):

    return self\.head == None

  def add\(self, item\):

    temp = Node\(item\)

    if self\.head == None:

      self\.tail = temp

    temp\.setNext\(self\.head\)

    self\.head = temp

  def append\(self, item\):

    temp = Node\(item\)

    if self\.head == None:

      self\.head = temp

      self\.tail = temp

    else:

      self\.tail\.setNext\(temp\)

      self\.tail = temp

  def size\(self\):

    current = self\.head

    count = 0

    while current \!= None:

      count = count \+ 1

      current = current\.getNext\(\)

    return count

  def search\(self, item\):

    current = self\.head

    found = False

    while current \!= None and not found:

      if current\.getData\(\) == item:

        found = True

      else:

        current = current\.getNext\(\)

    return found

  def search\_index\(self, idx\):

    current = self\.head

    found = False

    curr\_idx = 0

    while current \!= None and not found:

      if curr\_idx == idx:

        found = True

      else:

        current = current\.getNext\(\)

        curr\_idx \+= 1

    if found == False:

      raise ValueError\("Item not found"\)

    return current\.data



  def remove\(self, item\):

    current = self\.head

    previous = None

    found = False

    while current \!= None and not found:

        found = True

      self\.head = current\.getNext\(\)

    else:

      previous\.setNext\(current\.getNext\(\)\)

    if current\.getNext\(\) == None:


Поиск также O\(n\), добавление также O\(n\)\. Удаление O\(n\), но отличие двусвязного списка в том, что удаление с конца O \(1\), также, как и удаление из начала \(удаление из начала в односвязном списке O\(n\)\)\.

  def \_\_init\_\_\(self\):

    self\.\_head = None

    self\.\_tail = None

    self\.\_length = 0

  def is\_empty\(self\):

    return self\.\_head == None

  def size\(self\):

  def append\(self, item\):

    temp = DoubleNode\(item\)

    if self\.is\_empty\(\):

      self\.\_head = temp

      self\.\_tail = temp

    else:

      self\.\_tail\.set\_next\(temp\)

      temp\.set\_prev\(self\.\_tail\)

      self\.\_tail = temp

    temp = DoubleNode\(item\)

      self\.\_head = temp

    else:

      self\.\_head\.set\_prev\(temp\)

      temp\.set\_next\(self\.\_head\)

      self\.\_head = temp



    self\.\_length \+= 1



  def \_get\_node\(self, index\):

    if index >= self\.\_length or index < \(\-1\)\*self\.\_length:

      raise Exception\(f"list index \{index\} out of range"\)

    elif index >= 0:

      current = self\.\_head

      item\_counter = 0

      while current and item\_counter < index:

        current = current\.get\_next\(\)

        item\_counter \+= 1

    else:

      current = self\.\_tail

      item\_counter = \-1

      while current and item\_counter > index:

        current = current\.get\_prev\(\)

        item\_counter \-= 1



    return current



  def \_\_getitem\_\_\(self, index\):

    return self\.\_get\_node\(index\)\.get\_data\(\)



  def find\_node\_by\_data\(self, data\):

    current = self\.\_head

    found = False

    while current \!= None and not found:

      if current\.getData\(\) == data:

        found = True

      else:

        current = current\.get\_next\(\)

    return found



  def pop\(self, index\):

    if index >= self\.\_length or index < \(\-1\)\*self\.\_length:

      raise Exception\(f"list index \{index\} out of range"\)

    \# empty list

    elif self\.\_head == None:

      raise Exception\("You can't pop from an empty list"\)

    \# list of one element

    elif self\.\_head == self\.\_tail and \(index == 0 or index == \-1\):

      self\.\_length \-= 1

      temp = self\.\_head

      self\.\_head = None

      self\.\_tail = None

      return temp\.get\_data\(\)

    \# popping the first item

    elif index == 0 or index == self\.\_length\*\(\-1\):

      self\.\_length \-= 1

      temp = self\.\_head

      self\.\_head\.get\_next\(\)\.set\_prev\(None\)

      self\.\_head = self\.\_head\.get\_next\(\)

      return temp\.get\_data\(\)

    \# popping the last item

    elif index == \-1 or index == self\.\_length \- 1:

      self\.\_length \-= 1

      temp = self\.\_tail

      self\.\_tail\.get\_prev\(\)\.set\_next\(None\)

      self\.\_tail = self\.\_tail\.get\_prev\(\)

      return  temp\.get\_data\(\)

    \# popping from the middle

    else:

      current = self\.\_get\_node\(index\)

      current\.get\_prev\(\)\.set\_next\(current\.get\_next\(\)\)

      current\.get\_next\(\)\.set\_prev\(current\.get\_prev\(\)\)

      self\.\_length \-= 1

      return current\.get\_data\(\)



  def \_\_repr\_\_\(self\):

    current = self\.\_head

      list\_repr \+= f"\{current\.get\_data\(\)\} <\-> "

      current = current\.get\_next\(\)

    list\_repr \+= "None"

    return list\_repr

class Stack:

  def \_\_init\_\_\(self\):

    self\.items = \[\]

    if self\.is\_empty\(\):

    self\.top = None

    self\.size = 0

      raise IndexError\("Pop from empty stack"\)

    return value

      raise IndexError\("Peek from empty stack"\)

  def is\_empty\(self\):

  def \_\_str\_\_\(self\):

    current = self\.top

    while current \!= None:

      result \+= f"\{current\.value\} \-> "

      current = current\.next

    result \+= "None"

    return result

  def sift\_up\(self, i\):

    while i > 0:

      j = \(i \- 1\) // 2

      \# Проверяем на нарушение свойств кучи

      if self\.nodes\[i\]\.key > self\.nodes\[j\]\.key:

        self\.nodes\[i\], self\.nodes\[j\] = self\.nodes\[j\], self\.nodes\[i\]

      else:

        break

      \# Перескакиваем на место родителя

      i = j

  def sift\_down\(self, i\):

    \# Кол\-во узлов в куче

    n = len\(self\.nodes\)

    while True:

      left\_j = 2\*i \+ 1

      right\_j = 2\*i \+ 2

      largest = i

      if left\_j <= n \- 1 and self\.nodes\[left\_j\]\.key > self\.nodes\[largest\]\.key:

        largest = left\_j

      if right\_j <= n \- 1 and self\.nodes\[right\_j\]\.key > self\.nodes\[largest\]\.key:

        largest = right\_j

      if largest \!= i:

        self\.nodes\[i\], self\.nodes\[largest\] = self\.nodes\[largest\], self\.nodes\[i\]

        i = largest

      else:

        break

1. **Худший случай (Worst Case): O(n²)** — когда массив отсортирован в обратном порядке. Для каждого нового элемента алгоритму приходится сравнивать и сдвигать все предыдущие элементы.
2. **Средний случай (Average Case): O(n²)** — в среднем для случайного массива количество сравнений и перестановок составляет примерно n²/4, что всё равно даёт квадратичную зависимость.
3. **Лучший случай (Best Case): O(n)** — если массив уже отсортирован, алгоритм просто проходит по нему один раз. На каждом шаге делается одно сравнение и никаких перемещений.

- **Для маленьких массивов:** благодаря простоте и низким накладным расходам сортировка вставками часто работает быстрее, чем QuickSort или MergeSort. Поэтому во многих стандартных библиотеках (например, в Java или Python — Timsort) гибридные алгоритмы переключаются на вставки, когда размер подмассива становится меньше 10–20 элементов.
- **Для больших массивов:** при росте n время выполнения растёт квадратично. Если увеличить массив в 10 раз, время работы увеличится примерно в 100 раз, что делает алгоритм неэффективным для огромных объёмов данных.

**Решение.** Найдём $\mathbf{E}(\xi_k)$, $\mathbf{Var}(\xi_k)$, $\mathbf{E}(S_n)$, $\mathbf{Var}(S_n)$.

Распределение $\xi_k$ при $k = 1, \ldots, 4000$:

| $\xi_k$ | 0 | 1 000 000 | 250 000 |
|---|---|---|---|
| P | 0,9955 | 0,0005 | 0,004 |

Распределение $\xi_k$ при $k = 4001, \ldots, 10000$:

| $\xi_k$ | 0 | 1 000 000 | 250 000 |
|---|---|---|---|
| P | 0,9975 | 0,0005 | 0,002 |

При $k = 1, \ldots, 4000$ (верхний индекс указывает, для какой группы производятся расчёты):

$$\mathbf{E}(\xi_k^{(1)}) = 1\,000\,000 \cdot 0{,}0005 + 250\,000 \cdot 0{,}004 = 1500$$

$$\mathbf{Var}(\xi_k^{(1)}) = 10^{12} \cdot 0{,}0005 + 25^2 \cdot 10^8 \cdot 0{,}004 - 1500^2 = 747\,750\,000$$

При $k = 4001, \ldots, 10000$:

$$\mathbf{E}(\xi_k^{(2)}) = 1\,000\,000 \cdot 0{,}0005 + 250\,000 \cdot 0{,}002 = 1000$$

$$\mathbf{Var}(\xi_k^{(2)}) = 10^{12} \cdot 0{,}0005 + 25^2 \cdot 10^8 \cdot 0{,}002 - 1000^2 = 624 \cdot 10^6$$

Отсюда:

$$\mathbf{E}(S_n) = \sum_{k=1}^{n} \mathbf{E}(\xi_k) = 4000 \cdot 1500 + 6000 \cdot 1000 = 12 \cdot 10^6$$

$$\mathbf{Var}(S_n) = \sum_{k=1}^{n} \mathbf{Var}(\xi_k) = 4000 \cdot 747\,750\,000 + 6000 \cdot 624 \cdot 10^6 = 6735 \cdot 10^9$$

**1. Принцип нетто-премии.** Премия для $k$-го договора при $k = 1, \ldots, 4000$:

$$P_k^{(1)} = \mathbf{E}(\xi_k^{(1)}) = 1500$$

Премия для $k$-го договора при $k = 4001, \ldots, 6000$:

$$P_k^{(2)} = \mathbf{E}(\xi_k^{(2)}) = 1000$$

Отсюда:

$$\mathbf{E}(S_n) = \sum P_k = 12 \cdot 10^6 = u$$

$$R(u) = \mathbf{P}(S_n > u) = \mathbf{P}\!\left(\frac{S_n - \mathbf{E}(S_n)}{\sqrt{\mathbf{Var}(S_n)}} > 0\right) \approx \Phi(0) = \tfrac{1}{2}$$

$$x_\alpha = x_{0{,}95} = 1{,}64485$$

$$\lambda = \frac{x_\alpha \sqrt{\mathbf{Var}(S_n)}}{\mathbf{E}(S_n)} = \frac{1{,}64485 \sqrt{6735 \cdot 10^9}}{12 \cdot 10^6} \approx 0{,}3557$$

Индивидуальная защитная страховая надбавка $l_k = \lambda \mathbf{E}(\xi_k)$:

- при $k = 1, \ldots, 4000$:  $l_k^{(1)} = 0{,}3557 \cdot 1500 = 533{,}588$

$$P_k^{(2)} = (1 + \lambda)\mathbf{E}(\xi_k) = 1000 + 355{,}59 = 1355{,}73$$

Вот здесь есть решение [https://scinetwork\.ru/disk/file/66644](https://scinetwork.ru/disk/file/66644) \(Кириллица руководство к решению задач\)

**Пример 4.1.1.** Решено за 8 лет создать фонд в сумме 800 тыс. $ путём равных годовых платежей постнумерандо, на поступающие платежи начисляется 8,75% годовых. Четыре года платежи в фонд выплачивались согласно намеченному графику. Затем, в силу некоторых обстоятельств, в течение последующих лет платежи в фонд не поступали. После двух лет перерыва в выплатах в фонд решено вкладывать в конце каждого полугодия по 50 тыс. $. Выплаты в фонд прекращаются, как только накопленная сумма фонда превысит 800 тыс. $. На сколько лет больше придётся создавать фонд по отношению к намеченному сроку (8 лет)?

**Решение.** Согласно намеченному графику годовые платежи:

$$R = \frac{S}{s_{n;i}} = \frac{800 \cdot 0{,}0875}{1{,}0875^8 - 1} = 73{,}19925 \text{ тыс. \$}$$

К концу четвёртого года была накоплена сумма:

$$S_1 = R \cdot s_{4;8{,}75} = 73{,}19925 \cdot \frac{1{,}0875^4 - 1}{0{,}0875} = 333{,}51737 \text{ тыс. \$}$$

После изменения намеченного графика выплат фонд будет создаваться в течение $k$ лет, причём $2k$ должно быть целым числом. Исходя из условий примера, имеем неравенство:

$$S_1 \cdot 1{,}0875^{k-4} + 100 \cdot \frac{1{,}0875^{k-6} - 1}{s_{2;i}^{(2)}} \geq 800$$

или $333{,}51737 \cdot 1{,}0875^{k-4} + 50 \cdot \frac{1{,}0875^{k-6} - 1}{0{,}0875} \geq 800$. При $k = 8{,}5$ данное неравенство не выполняется, а при $k = 9$ выполняется. Фонд будет создан на год позже.
